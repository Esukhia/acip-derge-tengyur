༄##། །རྒྱ་གར་སྐད་དུ། བཛྲ་རྩརྩི་ཀ་སཱ་དྷ་ནཾ། བོང་སྐད་དུ། རྡོ་རྗེ་ཙརྩི་ཀའི་སྒྲུབ་ཐབས། རྡོ་རྗེ་ཙ་ཙི་ཀ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་གསུངས་པའི་ཆོ་ག་འདིས་སྟོང་པ་ཉིད་དུ་བསྒོམས་པའི་རྗེས་ལ་ཆུ་སྐྱེས་འདབ་མ་བརྒྱད་་པའི་སྟེང་དུ་ཉི་མ་ལ་གནས་པའི་ཧཱུཾ་ཡིག་ལས་སྐྱེས་པའི་རྡོ་རྗེའི་ལྟེ་བ་ལས་ཙཾ་ཡིག་གནས་པར་བསམས་ནས་དེ་ཡོངས་སུ་གྱུར་པ་ལས་རྡོ་རྗེ་ཙ་རྩི་ཀ་ཞལ་གཅིག་པ་སྤྱན་གསུམ་པ། སྐྱིལ་ཀྲུང་ཕྱེད་པའི་གར་སྟབས་ཀྱིས་རོའི་གདན་ལ་བཞུགས་པ། སྐུ་ལུས་རིད་ཅིང་མཆེ་བ་གཙིགས་པ་འཇིག་ས་པ་མི་མགོའི་ཕྲེང་་
