རྣམས་རྫོགས་སོ།། །། ན་ཌུ་ཧྲའི་སི་ལའོ། །ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱིའོ། །དྷི་ལ་ཨ་མྲའོ། །བས་རྣམས་ཀྱི་ར་ས་ལའོ། །དྲ་སྔོན་མའི་གྷྲའོ། །ཛྷ་བྷའི་མཐའ་ལས་སྔོན་མའི་ཛ་བ་རྣམས་ནི་ཛྷ་བྷ་སྟེ་ས་དྷྭ་དག་ལའོ། །གནས་མེད་པ་ལ་ཡང་ནའོ། །ཧྲའི་ཌྷྲའོ། །དྲུ་ཧྲ་སོགས་ཀྱི་གྷྲ་་ཉིད་ཌྷྲ་ཉིད་ཡང་ནའོ། །ཤས་སོགས་དབྱངས་ལ་བཱཧྲའི་བའི་ཨཽའོ། །ས་ཧའི་སའི་ཌྷ་ལ་ཥ?འོ། །ཙ་ཏུར་གྱི་ཤི་ལ་ཨཱཾ་ཡང་ངོ ། །རའི་གྲངས་ལའོ། །དྷི་མ་ཡིན་པ་ལ་མིང་གི་ན་དབྱིའོ། །ཛ་ཉ་དག་གི་ཛྙའོ། །ཤཝ་ལ་སོགས་པའི་ཡང་ངོ ། །ལྔ་ལ་ཨིཏྲ་ཀྱི་ཨ་ཏྲའོ། །པ་ཐིན་གྱི་ནུཊ?འོ། ། སི་ལ་ཨཱའོ། །པ་ཐིན་རྣམས་ཀྱི་ཊིའོ། །ིནར་རྣམས་ཀྱི་ཤི་དང་སི་ལའོ། །ཧ་ནར་གྱི་ནར་ལ་གྷའོ། །ཛས?་ཤས?་དག་ནི་འཇིག་གོ་།ཥ་ཎ་ལས་སོ། །ཥྚ་ནར་ལས་ཊཽ་ཡང་ནའོ། །འདི་རྣམས་ལ་ཨཱ་ཡང་ནའོ། །ི་དཾ་གྱི་སྐྱེས་པ་ལ་ཨ་ཡཾ་མོ། །དའི་མའོ། །ཊཱ་ཨོ་སྲ་དག་ལ་ཨ་ནའོ། ས་བྷ་ལ་ཨའོ། །བྷིས?་བྷིས?འོ། །ཚ་ཤ་ཥ་རཱ་ཛྲ་ལ་སོགས་པའི་ཥའོ། །ཥའི་ཌའོ། །ཏའི་སའོ། །ཙུའི་ཀུའོ། ། ཙ?འི་དབྱི་ཞིང་རིང་པོ་འང་ངོ ། །ཏི་རཤྩ་ལ་སོགས་པའོ། །ུ་རྀ་ཏྲ་དག་ལས་ནུམ?འོ། །ན?་ས?་ཨཔ?་མ་ཧ་ཏ?་ཀྱི་དྷི་མ་ཡིན་པ་ལ་རིང་པོ་སྟེ་ཤི་ལའང་ངོ ། ། ཏུ་དང་སའི་སི་་ལའོ། །ཌའི་ཎའོ། །དོཥྲ་\་ཌཱོཤཱ\?\རྣམས་ཀྱི་རའོ། །ཕུམྲ་སའི་ཨ་སུང་ངོ ། །དབྱངས་ལ་མའོ། །མི་འབྱུང་བ་ལ་པུམ་སའི་སུ་ལ་ཀ་ཀྲའོ། །སྲ་ཀྲ་དག་གི་སྔོན་མའོ། །བས་ཀྱི་བ་ཨུའོ། །ུ་ཤ་ནས་རྣམས་ཀྱིའོ། །སི་ལ་སའོ། །སིའི་ཨཽའོ། །མ་ལས་ཨུ་དག་གོ་།མང་པོ་ཉིད་ལ་ཨེའི་ཨཱིའོ། ། མཚུངས་པ་ལ་ཨ་ད་སྲ་ལས་ཀ་སྟེ་སི་སོགས་བཞིན་ནོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཧ་སའི་མཐའ་ལས་སྐྱེས་པའི་རྟགས་རྣམས་རྫོགས་སོ།། །།ན་ཧྲའི་དྷྲའོ། །དི་པྲ་ཀྱི་སི་ལ་ཨཽའོ། ། མྲ་ལ་ཨཱ་ཡང་ནའོ། །ར་ས་ལ་ཨུའོ། །ཏྲི་ཙ་ཏུ་ར?དག་གི་བུད་མེད་ལ་ཏྲི་སྲྀ་ཙ་ཏ་སྲྀ་རཱྀ་བཞིན་ནོ།། ར?་བ?་དག་ལ་ཨི་ཨུ་དག་གི་ཧ་ས་ལའོ། །བུད་མེད་ལ་ཨི་ཡམ་མོ། ། པྲ་རྣམས་ཀྱི་བྷ་ལ་དྲའོ། །དི་ཤྲ་རྣམས་ཀྱི་ཀའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཧ་སྲའི་མཐའ་ལས་བུད་མེད་ཀྱི་རྟགས་རྣམས་རྫོགས་སོ།། །། ཧ་ནར་གྱི་སའོ། །ན་དབྱིའོ། །ན་མཐའ་ཨ་ཏྲ་ངི་ཤི་དག་གི་་
༄#། །ཡང་ན་དབྱི་བ་སྡེ་བ་སྦྱོར་ལའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཧ་སའི་མཐའ་ལས་མ་ནིང་གི་རྟགས་རྣམས་རྫོགས་སོ།། །།སི་ཡིས་ཏྭཾ་ཨ་ཧཾ་ངོ ། གཉིས་ཚིག་ལ་ཡུ་བ་ཨཱ་བ་དག་གོ་།ཽ་ཨཱ་མ?འོ། །ཛས་ཀྱི་ས་ཡཱུ་ཡཾ་བ་ཡཾ་མོ། །གཅིག་ཉིད་ལ་ཏྭ་དྲ་་མདྲ་དག་གོ་། མྲ་ས་བྷི་ལ་ཨཱའོ། །ཤསཝ་ཀྱི་ནའོ། །ཊཱ་དི་དག་ལ་ཨེའོ། །ངེ་ཡིས་ཏུ་བྷྱ་མྲ་མ་ཧྱོཾ་ངོ ། །བྷྱ་སྲ་ཤྦྷྱ་མྲ་མོ། །ང་སི་བྷྱ་སྲ་དག་གི་ཤྟུའོ། །ངསྲ་ཀྱི་ཏ་བ་མ་མའོ། །སཱཾ་ཨཱ་ཀཾ་ངོ ། །ཡུཥྨ་དྲ་ཨ་སྨདྲ་དག་ལ་དྲུག་པ་བཞི་པ་གཉིས་པ་རྣམས་ཀྱི་ཏེ་མེ་བམྲ་ནཽ་བསྲ་ནསྲ་དག་གོ ། མྲ་གྱི་ཏྭཱ་མཱ་དག་གོ་།དང་པོ་ལ་མ་ཡིན་ནོ། །ཙ་ལ་སོགས་པ་རྣམས་ཀྱིས་ཀྱང་ངོ ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཡུཥྨ་དྲ་ཨ་སྨྲད་ཀྱི་ཆོ་ག་རྣམས་རྫོགས་སོ།། །།ཙ་ལ་སོགས་པ་ནི་པཱ་ཏའོ། །ཀི་མྲ་ལས་སྤྱི་ལ་ཙི་ཏྲ་ལ་སོགས་པའོ། །པྲ་ལ་སོགས་པ་ཉེར་བསྒྱུར་རོ། ། བྱིངས་ལ་ཐོག་མའོ། །དེ་རྣམས་མི་ཟད་པའོ། །ཀྟྭཱ་ལ་སོགས་པའི་མཐའ་རྣམས་ཀྱང་ངོ ། །མི་ཟད་པ་ལས་རྣམ་དབྱེ་རྣམས་འཇིག་གོ་།ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་མི་ཟད་པའི་སྡེ་རྣམས་རྫོགས་སོ།། །། ཛ་སོགས་ལས་ཀྱང་ཨཱ་པྲའོ། །ཀཱ་པྲ་ལ་ཨ་ཏྲ་ཀྱི་ཨིའོ། །ཐུང་་དུ་ཡང་ནའོ། །ན་རྀ་ཨཎྲ་ལ་ཨཱི་པྲའོ། །ི་ཨ་དག་གི་དབྱིའོ། །ཥ་ཊ?་ཨུ་རྀ་ཏྲ་ལས་སོ། །ན་ད་ལ་སོགས་པ་ལས་སོ། །ིནད་སོགས་ལས་ཨཱ་ནཱི་པྲ་ཡང་ངོ ། །ུ་པ་དྷཱ་ཡ་མ་ཡིན་པ་ལས་རིགས་ལའོ། །རང་ལུས་ལ་ཡང་ནའོ། །མ་ནུ་སོགས་ཀྱི་ཨཻ་ཡང་ངོ ། །པཏྣྱ་ལ་སོགས་པ་་ལས་སོ། །པྲ་ཏཱི་ཙཱི་ལ་སོགས་པའི་སྒྲ་ནི་ནི་པཱ་ཏའི་མཐའ་འོ། །ུ་ལས་ཡོན་ཏན་ལ་ཡང་ནའོ། །ུཏྲ་ལས་ཨུའོ། །ཡུ་བནར་ལས་ཏིའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་བུད་མེད་ཀྱི་རྐྱེན་རྣམས་རྫོགས་སོ།། །།རྟགས་ཀྱི་དོན་ལ་དང་པོའོ། །བོད་པ་ལ་འང་ངོ ། །བྷོས་་རྣམས་སོ། །ལྷག་མ་རྣམས་བྱ་བའི་ལས། བྱེད་པོས་བསྒྲུབ་པའི་ཐབས། སྦྱིན་པའི་གནོད། བྱེ་ཞིང་ལྷུང་བ། འབྲེལ་བའི་རྟེན་དུ་གྱུར་བའི་དངོས་པོ་དག་ལའོ། །དུས་དང་ལམ་པར་མཚམས་མེད་པ་ལའོ། །བྱེད་པ་པོ་གཙོ་བོས་བྱ་བ་ནུས་པའི་སྒྲུབ་བྱེད་ལའོ། །མི་བྱེད་པ་དང་ལྷན་་
ཅིག་པ་དང ། ཕྱག་འཚལ་བ་དང ། བོར་བ་དང་ལོགས་སུ་བཀར་བ་དང་རྗེ་བོ་ལ་སོགས་པ་རྣམས་ལས་སོ། །ཀྟ་ལ་སོགས་པ་མ་ཡིན་པའི་ཀྲྀཏྲ་ཀྱི་བྱེད་པོ་དང་ལས་དག་ལ་དྲུག་པའོ། །དྲན་པའི་ལས་ལའང་དྲུག་པའོ། །འཇིགས་པའི་རྒྱུ་ལའང་ལྔ་བའོ། །རྒྱུ་་དང་རབ་ཏུ་སྦྱོར་བ་ལ་དྲུག་པའོ། །འདི་འདྲར་ཡོད་པ་ལ་གསུམ་པའོ། །གང་གི་ལུས་ཀྱི་རྣམ་འགྱུར་ལའོ། །སྐྱེད་པར་བྱེད་པའི་རང་བཞིན་ལའོ། །ཱ་ང་ལ་སོགས་པ་སྦྱོར་བ་ལ་ལྔ་པའོ། །འཁྲུགས་པ་ལ་སོགས་པའི་སྦྱོར་བ་ལའང་ངོ ། །མ་གུས་པ་ལ་དྲུག་པ་དང་བདུན་པ་་དག་གོ་།གཞན་ལས་བཤད་པའི་དང་པོའོ། །སྡེབ་སྦྱོར་ལས་ནི་སི་སོགས་ཐམས་ཅད་དུའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་བྱེད་པ་ཅན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། !་།།མིང་རྣམས་ཀྱི་ཕན་ཚུན་བསྡུ་བ་ཡང་ངོ ། །དེ་ཡང་དྲུག་ཏུ་དབྱེའོ། །མི་ཟད་པ་་སྔ་མ་ནི་མི་ཟད་པར་གྱུར་བའོ། །བསྡུ་བ་དང་རྐྱེན་དག་ལའོ། །དེ་ནི་མ་ནིང་ངོ ། །མི་ཟད་པར་གྱུར་པ་ལས་སོ། །མཚུངས་པ་མ་ཡིན་པ་ལ་ཡ་ཐཱའོ། །ཏྲ་ལས་ཨཱཏྲ་མ་ཡིན་པའི་ཨམྲའོ། །ཊཱ་ངི་དག་ཡང་ནའོ། །ལོགས་སུ་བཀར་བའི་དོན་ལ་ཡཱ་བཏྲ་ཡང་ངོ ། ། མ་ལ་་སོགས་པ་དེའི་སྐྱེས་བུའིའོ། །པཱ་ནའི་ཡང་ནའོ། །ན་ཉ?་ལའོ། །ན་ཨའོ། །དབྱངས་ལ་ཨནརའོ། །དང་གི་དོན་ལ་ཟླས་དབྱེའོ། །ཕྱོགས་གཉིས་པ་དང་ཟླས་དབྱེ་ལ་གཅིག་ཉིད་དོ། །གྲངས་ཐོག་མ་ཅན་ནི་ཕྱོགས་གཉིས་པའོ། །ཕྱོགས་གཉིས་པ་ལ་མཉམ་དུ་འདུས་པའི་ཨཏྲ་་ལ་ཨཱིཔྲའོ། །གཞན་གྱི་དོན་ལ་འབྲུ་མང་པོའོ། །ིནྡུ་ལ་སོགས་པ་དག་ལས་ཀྱང་མ་ཡིན་ནོ། །པྲ་ཛ་མི་དྷ་དག་ལ་ཨ་སུ་ཀྲའོ། །དྷརྨ་ལས་ཨནརའོ། །གཞན་གྱི་དོན་ལའོ། །སྐྱེས་པ་བཞིན་ཡང་ནའོ། །གོ་ཡིའོ། །ཊ་ཨཌ་ཀ་རྣམས་སོ། །ནའི་ཡང་ནའོ། །མཉམ་པའི་དོན་ལ་ལས་་འཛིན་པའོ། །མིང་དང་ཀྲྀཏྲ་ལས་ཀྱང་བསྡུ་བའོ། །ས་ཧ་ལ་སོགས་པ་རྣམས་ས་ལ་སོགས་པའོ། །ཀུའི་ཀཏྲའོ། །སྐྱེས་བུ་ལ་ཡང་ནའོ། །འགའ་ཞིག་ཏུ་དབྱི་བ་མ་ཡིན་ནོ། །ཟླས་དབྱེ་ལས་དང་པོ་ཡང་ངོ ། །ཕྱོགས་དང་གྲངས་ཀྱི་མིང་ཅན་ལའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་་

༄#། །ཥ་ཌྷ་དག་གི་གའོ། །ཧ་ན?་རི་ཏ?་ལས་སྱབ?་ལའོ། །པ?་མེད་བ་ལ་སའི་ས?་ལ་ཏྲའོ། །ུ་པ་དྷཱ་ཐུང་ངུའིའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་འབྱུང་འགྱུར་གྱི་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱ་བ་ཤོར་བ་ལ་སྱབ?འོ། །ཞེས་པ་བརྡ་་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་བྱ་བ་ཤོར་བའི་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱུང་ཟིན་པ་ལ་སིའོ། །པ་ལ་དཱ་ལ་སོགས་པ་ལས་སོ། །སི་ལ་ཨཱ་དང་བིད་ལས་སོ། །ུས་ལ་ཨཱ་དབྱིའོ། །མཱ་ལས་ཨཊ?འིའོ། །དྷ་ལའང་སི་དབྱིའོ། །པ་ལ་ཎ་འགྲོ་བའོ། །སི་ལའོ། ། ིཊ?་ལས་ཨཱིཊ?་ལའོ། །ཐུང་དུ་ལས་ཛྷ་ས་ལ་དབྱིའོ། །ིང་གི་སི་ལ་གཱི་ཡང་ནའོ། །ཛྷ་ས་ལས་སོ། །ིར་ལ་ཡང་ནའོ། །ིཊ?མེད་པའི་ནཱ་མི་ལས་འཕེལ་བའོ། །ཛྷ་ས་ལ་དྲྀ་ཤ་རྣམས་ཀྱི་ཨར་ནི་ར?འོ། །སི་སཱི་དག་ལའོ། །ལྀ་འགྲོ་བ་དང་བུ་ཥ་སོགས་ལ་ངའོ། །ཧ་་ཤ་ཥ་རྣམས་ཀྱི་མཐའ་ལས་ས་ཀྲའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་བྱུང་ཟིན་གྱི་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱེད་པ་པོ་མ་ཡིན་པ་ལས་ཏནར་ལ་ཨིཎ?འོ། །དབྱིའོ། །ུ་པ་དྷཱ་ཐུང་དུ་ལས་སོ། ། ཏྲ་ལས་ཡུ་ཀ?འོ། །ཞེས་པ་བརྡ་སྤྲོད་་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་དངོས་པོ་དང་ལས་ཀྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།ཧ་ས་དང་པོ་ལས་ཕུལ་དུ་བྱུང་བ་ལ་ཡང?སྟེ་གཉིས་ཀྱང་ངོ ། །ཡང?་ལའོ། །དེ་བྱིངས་སོ། །ཱ་ཏྲའོ། །དབྱངས་དང་པོའི་ཕ་རོལ་ལོ། །ཉ་མ་དང་ཛྷ་པ་རྣམས་ལ་ནུ་ཀྲའོ། །ན་ནི་་དབྱིའོ། །རྀ་ཏྲ་ཨུ་བ་དྷཱའི་རཱི་ཀྲའོ། །ཙ་ར་ཕ་ལ་དག་གི་ཨའི་ཡང་ཨུཏྲའོ། །རྀཏ?་ཀྱི་རིའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཡང?་རྐྱེན་གྱི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།གཞན་ལ་འཇིག་པ་ཡང་ན་རྗེས་སུ་འཇུག་གོ་།པ?་མེད་པའི་ཧ་ས་ལས་ཀྱང་ངོ ། ། གཉིས་ལས་བྱིངས་པ་འགྲོ་བ་ལ་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཞེས་པ་བརྡ་སྤྲོད་བ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཡང?་འཇིག་པའི་མཐའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱེད་པ་པོ་ལ་ཡང་ངོ ། །ཉེ་བར་འཇལ་བའི་ཀུན་ཏུ་སྤྱོད་པ་ལ་ཡའོ། །བདག་ཉིད་འདོད་པ་ལ་སའོ། །ུ་རྀའི་་
ས་ལའོ། །ས་ཨིཊ?་མེད་པ་ལ་མ་ཡིན་ནོ། །ས་ལ་རིང་པོའོ། །ས་ལ་ཨ་ཨིའོ། །ཧན?་ལས་སཎ་འགྲོ་བའོ། །རྀཏ?་ཨིར?འོ། །ཡའི་ཨཏྲའོ། །ས་ལ་ཨི་ས?འོ། །ཱ་པྣོ་ཏིའི་ཨཱིའོ། ། ཤ་ལ་ཨནཱ་ཡ་ཡང་ནའོ། །གུབྲ་རྣམས་ལས་སོ། །ཱ་ཡའོ། །མིང་ལ་ཡ་སྟེ་ཨའི་ཡང་ཨཱིའོ། ། བྱེད་པ་ལ་འང་ཡའོ། །བྱེད་པ་པོ་ལ་ཉི་ཌ་འགྲོ་བའོ། །ཉི་ཉིད་ལ་ཐུང་ངུའོ། །པྲྀ་ཐ་སོགས་ཀྱི་རའོ། །ཧན་གྱི་གྷཏྲའོ། །ས་དའི་ཤཏྲའོ། །རྀ་ཨཱ་ཏྲ་ལས་ཉི་ལ་བུ་ཀ?འོ། །ི་ཨཱའོ། །ཙུ་ར་སོགས་ལ་ཉིའོ། །ཉི་དང་ཨང་ལས་གཉིས་ཀྱང་ངོ ། །ཉིའིའོ། །ཐུང་དུ་རིང་པོའོ། །ང?་ལ་ཨུ་པ་དྷཱ་་ཐུང་ངུ་ནི་ཐུང་ངུའོ། །རྀཏ?་ཀྱི་མ་ཡིན་ནོ། །བཧ་ལས་ཀྟིའོ། །ང?་ལ་སྔོན་མའི་ཌྷ་ནི་ཛའོ། །ཀྲྀ་པའི་ར?་ལའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་རྐྱེན་དང་སྦྱོར་བའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།དྲྀ་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་པ་ཤྱ་ལ་སོགས་པ་རྣམས་སོ། །གམ?་རྣམས་ཀྱི་ཚའོ། །ཛཉཱ་ཛནར་དག་གི་ཛཱའོ། །ཤྲུའི་ཤྲྀའོ། །པཱུ་ལ་སོགས་པའི་ཐུང་ངུའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཀུན་ནས་བསྒྱུར་བའི་རབ་ཏུ་བྱ་བ་རྣམས་རྫོགས་སོ།། །།ནི་ལས་བི་ཤ་སོགས་ལའོ། །བི་པ་རཱ་དག་ལས་ཛི་ལའོ། །ཱང?ལས་དཱ་ལ་ཁ་གདངས་པ་་མ་ཡིན་པ་ལའོ། ། ནུ་སཾ་པ་རི་རྣམས་ལས་ཀྱང་ཀྲཱིཌ?་ལའོ། །སམ?་ཨ་བ། པྲ། ཨུ་པ། བི་རྣམས་ལས་སྠཱ་ལའོ། །ུཏྲ་བི་དག་ལས་ཏ་པ་ལའོ། །ཱང?་ལས་ཡ་མ་ཧནར་ལའོ། ། ནུ་དཱཏྟ་རྣམས་དབྱི་བ་ཉིད་དོ། །ུཏྲ་ལས་ཙ་ར་ལ་འདོན་པ་ལའོ། །པི་ཨ་བ་པ་རི་རྣམས་ལས་ཀྲཱི་་ལའོ། །ཛྙཱ། ཤྲཱུ་སྨྲྀ་དྲྀ་ཤྲ་རྣམས་ས་མཐའ་ཅན་ལ་ཨཱཏྲའོ། །འཚེ་བ་སོགས་ལས་གཞན་པའི་ལས་གཅིག་ལ་འཇུག་པ་ལ་ཨཱཏྲའོ། །པི་ཨཱང?་པ་རི་ཨུ་པ་རྣམས་ལས་ར་མ་ལའོ། །བྲྀཏ?རྣམས་ལས་སྱཔྲ་དག་ལ་ཡང་ནའོ། །མུ་ཙ?་སོགས་ལ་མུ་མ?འོ། །ིཏྲ་འགྲོ་བ་ལ་མུ་མ?འོ། ཤམ?་རྣམས་ཀྱི་རིང་པོའོ། །ཡ་ལ་ཨོའོ། །པུའི་རྀ་ཨུར?འོ། །རུདྲ་སོགས་ལསྲ་བཞི་པོ་རྣམས་ཀྱི་ཧ་དང་བ་ས་ལའོ། །ད་རི་དྲཱ་སོགས་ཀྱི་ད་འགྲོ་བ་ལ་ཨིཏྲ་སྟེ་ཨཱ་དབྱི་བ་ཡང་ངོ ། །ཱིཌ?་ཨཱི་ཤྲ་དག་ལས་སོ། །ུ་ཨཽའོ། །ཡུ་ནུ་རུསྟུ་རྣམས་ལས་ཨཱིར?་ཡང་ནའོ། །ཏན?་
༄#། །རྣམས་ཀྱི་ཨ་ནུ་དཱཏྟ་དབྱི་བ་ཉིད་དོ། །ནམ?་ལའོ། །པྲ་ལ་རཉྫ?་དཾ་ཤྲ། །ཥཉྫ?་ཥྭཉྫ?་རྣམས་ཀྱིའོ། །བིད་ལས་ཏིཔ?་ལ་སོགས་པ་དགུ་རྣམས་ནི་ཎཔྲ་ལ་སོགས་པར་ཡང་ནའོ། །སི་ལ་སའོ། །ཛ་ཧི། ཨེདྷི། ཤཱ་དྷྲིའོ། །སྟི་ལས་ཨཱིཊ?འོ། །ཌྷའི་་ཌྷ་ལ་དབྱི་ཞིང་རིང་པོ་ཡང་ངོ ། །ས?་ཀ?་སོགས་ཀྱི་ཡང་ངོ ། །གཞན་དུ་ས?་ཛ?འོ། །དིཔ?་ལ་སའི་ཏ་སྟེ་སྲིཔ?་ལ་ཡང་ནའོ། །བ་འགྲོ་བ་ལ་མྲྀ་ཇིའི་འཕེལ་བའོ། །ནཱི་ཛ་རྣམས་ཀྱི་ཡོན་ཏན་ནོ། །སྔོན་མའི་རྀ་པྲྀ་དག་གི་ཨིའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཀུན་ནས་བཤད་་པའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱེད་པ་པོ་ལ་ཀྲྀཏྲའོ། །ཏྲི་བུཎ?་དག་གོ་།ཀྲྀཏྲ་ལའོ། །ཱུཏྲ་འགྲོ་བ་ལས་ཡང་ནའོ། །སཧྲ་བཧྲ་དག་གི་ཨ་ཡིག་ནི་ཨོཏྲའོ། །ཡུ་བུ་དག་གི་ཨ་ན་ཨ་ཀ་དག་གོ་།ནཱ་མིའི་ཨུ་པ་དྷཱ་ལས་ཀའོ། །པ་ཙི་ན་ནཱི་གྲ་ཧཱ་སོགས་ལ་ཨ་ཡུ་ཎི་་ནིའོ། །རུ་ལ་ཡུཎ?འོ། །དྲྀ་ཤྲ་སོགས་ལ་ཤའོ། །ཤ་འགྲོ་བ་ལ་བཞི་པོ་བཞིན་ནོ། །ཛྭ་ལ་སོགས་ལ་ཎའོ། །ལས་ལ་ཨཎ?འོ། །ཱཏ?ལས་ཌའོ། །མིང་ལ་འང་ངོ ། །ཤུ་ཙའི་དྲ་ལ་ཤཱུའོ། །ུ་རས?ཀྱི་དབྱི་ཞིང་མུ་མྲ་ཡང་ནའོ། །བི་ཧཱ་ཡ་སའི་བི་ཧ་འང་ངོ ། ། ཊ་དག་གོ། ཨི་ཁ་ཁིའོ། །ཁ་འགྲོ་བ་ལ་སྔོན་མའི་ཚིག་གོ་།ེ་ཛ?་རྣམས་ཀྱི་ཁ་ཤྲའོ། །བྱེད་པ་ལ་ཁྱུཊ?འོ། །བྷཛ?་རྣམས་ཀྱི་བིཎ?འོ། །བིའིའོ། །ཤས?་སོགས་དབྱངས་ལ་བ་ཧའི་བ་ཨཽའོ། །ས་ཧྲའི་སཱཌྷ་ལ་ཥ་འང་ངོ ། །བྱིངས་ཀྱི་མའི་ནའོ། །བྱུང་བ་མ་ཡིན་པ་ལས་ཀྲྀ་བྷཱུ་ཨསྲ་སྦྱོར་བ་ལ་་ཙཝིའོ། །ཙཝི་ལ་རིང་པོ་སྟེ་ཨའི་ཡང་ཨཱིའོ། །ཙཝི་ལ་ས་དབྱི་བ་འང་ངོ ། །ཱཏྲ་ལ་མ་ཎི་པྲ་ཀཝ་ནི་པྲ་བ་ནི་པྲའོ། །སྠཱ་རྣམས་ཀྱི་ཨཱིའོ། །ཀཝི་པྲའོ། །ཐུང་དུའི་ཀྲྀཏྲ་ཀྱི་པ་འགྲོ་བ་ལ་ཏུཀྲའོ། །ཀཝི་པའོ། །ཉེ་བར་འཇལ་བའི་བྱ་བ་ལ་དྲྀ་ཤྲ་ལ་ཊཀྲ་སཀྲ་དག་ཀྱང་ངོ ། །སརྦ་སོགས་ཀྱི་ཨཱའོ། །ཀིམ?་ཨི་དམྲ་དག་གི་ཀཱི་ཨཱི་ཤྲ་དག་གོ་།འདས་པ་ལ་ཎི་ནིའོ། །ཀཝི་པྲ་བ་ནི་པྲ་ཌ་རྣམས་སོ། །ཀྟ་ཀྟ་བ་ཏུའོ། །ལས་མེད་པ་ལས་འགྲོ་བའི་དོན་དང་བྱེད་པ་པོ་ལ་ཀྟའོ། །ིཊྲ་དང་བཅས་པའི་ཀྟ་ལ་ཡང་ནའོ། །ཀ་འགྲོ་བ་ལའོ། །ཱཏ?་ཨཱིཏྲ་འགྲོ་བ་ལའོ། །ཉི་ཨིཏྲ་འགྲོ་བ་ལ་ད་་
ལྟར་བ་ལ་ཏཀྲ་ཀྱང་ངོ ། །ད་ལས་ཏའི་ན་སྟེ་དའི་ཡང་ངོ ། །ཏྲ་ཀྱི་ཀ་འགྲོ་བ་ལ་ཛགྷྲའོ། །ར་ལས་སོ། །ལཱུ་སོགས་དང་ཨོཏྲ་འགྲོ་བ་ལས་སོ། །ཏ་ལ་དཱའི་ད་དྲའོ། །དབྱངས་ལས་ཏ་ཡང་ནའོ། །བས?་ཀྵུ་དྷྲ་དག་ལ་ཨིཊ?འོ། །གྷས?་སོགས་ཀྱི་ཥའོ། །ཏྲཎ་སོགས་ནི་ནི་པཱ་ཏ་ཡང་་ནའོ། །སམྲ་པ་རི་ཨུ་པ་རྣམས་ལས་ཀྲྀ་ལ་རྒྱན་གྱི་དོན་ལ་སུཊ?འོ། །དངོས་པོ་ལ་གྷཉྲའོ། །ཛནརའི་ཛཱའོ། །པཙ?་ལས་བའོ། །ཤུ་ཥ་ལ་ཀའོ། །ཀྵཱ་སོགས་ལ་མའོ། །སྤཱ་ཡའི་སྥཱིའོ། །ཀཝ་སུ་ཀཱ་ན་དག་ཎབྲ་ཨེ་བཞིན་ནོ། །བྱ་བ་ལ་ཤྀཏྲ་ཤཱ་ན་དག་སྟེ་ཏིཔྲ་ཏེ་བཞིན་ནོ། །ཱ་ན་ལས་་ཨཏྲ་ལ་མུཀྲའོ། །ཱས?་ལས་ཨཱ་ནའི་ཨཱིའོ། །ཏྲ་ལ་ཤ་ཏུ་སྟེ་ཨཱིཔ?་ལ་ཡང་ནའོ། །ཏྲ་ལས་ཨཔ?་ཀྱི་ཨ་ལ་རྟག་ཏུའོ། །པིད?་སོགས་ལས་བ་སུ་ཡང་ནའོ། །ངང་ཚུལ་ལ་ཏྲྀནའོ། །།ིཥྞུ་སྣུ་ཀྣུ་རྣམས་སོ། །ཥཱ་ཀ་ཨུ་ཨུ་ཀཎ?འོ། །ས་མཐའ་ཤཾ་ས་དག་ལ་ཨུ་ཡང་ངོ ། །ནམ?་སོགས་ལ་རའོ། །གྷས?་སོགས་ལ་ཀྨ་རའོ། །ཚི་དི་བྷི་དི་བི་ད་རྣམས་ལ་ཀུ་རའོ། །བྷཱས་སོགས་ལ་གྷུ་རའོ། །ཡང?་ལ་ཨཱུ་ཀའོ། །ཱཏྲྀ་རཏྲ་ལ་བྱུང་ཟིན་ལ་ཀི་སྟེ་གཉིས་ཀྱང་ངོ ། །ུཎ་ལ་སོགས་པ་ནི་རྟག་ཏུའོ། །མ་ནི་ལ་ཨུ་པ་དྷཱའི་རྀ་རའོ། །དྷུ་སོགས་ལ་ཨུ་ལིའོ། །ཱི་ཤ་་སོགས་ལ་བ་རའོ། །རཱཛ?་སོགས་ལ་ཨནརའོ། །ི་མན?་ཏ?་ཨ་སུཀ?འོ། །བཙ?་སོགས་ལ་ཨས?འོ། །མཆོག་མིན་གྱི་ལས་ལ་ཀེ་ལི་མའོ། །ཤཾ་པྲ་བི་རྣམས་ལས་བྷཱུ་ལ་ཌུའོ། །ཡུ་བཧྲ་ཨ་གི་རྣམས་ལས་ནི་པ?འོ། །ི་དི། ཙ་དི་ཤ་ཀི་རུ་དི་རྣམས་ལས་རའོ། །ུཎ་ལ་སོགས་པ་་དཔག་ཏུ་མེད་པའོ། །འབྱུང་འགྱུར་ལས་དེའི་དོན་ལ་ཏུ་མ?འོ། །དངོས་པོ་ལ་གྷཉྲའོ། །ཙ་ཛ་དག་གི་གྷ་འགྲོ་བ་ལ་ཀ་ག་དག་གོ་།བྱེད་པོ་མ་ཡིན་པའི་མིང་ལའོ། །དབྱངས་ལས་ཨའོ། །མདྲ་རྣམས་ལའོ། །གྷནར་ལས་སྲབ་ལའོ། །ཡི་གེ་བསྒྱུར་བ་ལ་འང་ངོ ། །ཊུ་འགྲོ་བ་ལས་་ཨ་ཐུའོ། །ཌུ་འགྲོ་བ་ལས་དེ་འདས་པ་ལ་ཏྲི་མྐྲའོ། །ནཊ?་ཀི་དག་གོ་།པྲཙྪ་བིཙྪ་དག་གི་ན་ལ་ཤའོ། །ཡུཊ?འོ། །ུཏྲ་ལས་སྠ་སོགས་ཀྱིས་དབྱིའོ། །སྒྲུབ་ཐབས་དང་རྟེན་དག་ལ་འང་ངོ ། །ཱིཥ?། དུཿ ་སུ་རྣམས་ལ་ཁལ་ཡཱུཊ?འོ། །ཏ་བྱ་ཨཱ་ནཱི་ཡ་དག་གོ་།དབྱངས་ལས་ཡའོ། །པུ་ཤ་་
