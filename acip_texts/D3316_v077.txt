༄##། །རྒྱ་གར་སྐད་དུ། བི་དྱཱ་དྷ་ར་པི་ཊ་ཀ་སཾ་ཀྵི་པྟ་མཉྫུ་ཤྲཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རིག་པ་འཛིན་པའི་སྡེ་སྣོད་དུ་གསུངས་པའི་མདོར་བསྡུས་པའི་འཇམ་དཔལ་གྱི་སྒྲུབ་པའི་ཐབས། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །སྔོན་དུ་བསྟན་པའི་ཆོ་གས་སྟོད་པ་ཉིད་བསྒོམས་པའི་རྗེས་་སུ་པདྨ་དང་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཐིག་ལེ་དཀར་པོ་རྣམ་པར་བསྒོམས་ནས་སྤྲོ་བ་ལ་སོགས་པ་སྔོན་དུ་འགྲོ་བས་དེ་ཡོངས་སུ་གྱུར་པ་ལས་ཡི་གེ་དྷཱིཿ དྐར་པོ་དེ་ཡོངས་སུ་གྱུར་པ་ལས་འཇམ་དཔལ་ཨ་ར་པ་ཙ་ན་གཅིག་པུ་འབའ་ཞིག་ཉིད་དོ། །སྙིང་གར་ས་བོན་རྣམ་པར་བསམས་ནས་ཡང་་
