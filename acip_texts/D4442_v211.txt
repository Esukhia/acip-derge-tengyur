ཌནྟི་ཨ་མ་ས་ན་བནྡྷེ། ར་ག་ཏི་ཨ་མ་ས་ན་བནྡྷེ། ཙཽ་ས་ཋི་ཡོ་གི་ནི་བནྡྷེ། བ་བ་ནི་པ་ར་བནྡྷེ། ཨ་ཋ་སི་ཀ་རུ་པ་བནྡྷེ། དེ་བནྡྷེ། ད་ནཽ་བནྡྷེ། ཀི་ཡོ་ཀ་ར་ཡོ་བནྡྷེ། ཨ་པ་ནོ་པ་ར་ཨོ་བནྡྷེ། ལ་གེ་ཨོ་པ་ལི་གེ་ཨོ་བནྡྷེ། ཚ་ཏི་ས་རོག་མེ་རོ་ག་གེ་།གོ་ཧ་ཛ་ཏེ། ཀ་པ་ལ་ཏེ། ཙོཾ་ཊི་ཏེ། ན་བོ་ན་རི་བ་ཧ་ཏ་རི་ཀོ་ཋ་ཏེ། ཡ་ཧཾ་པ་བ་ཧི་ཏཾ་བནྡྷེ། ཨ་དེ་བྷཻ་རོ་བནྡྷེ། ཡུ་ག་དེ་བཻ་རོ་བནྡྷེ། ཙ་རི་ཁུ་ཊ་པཉྩ་ཨི་དེ་ཧ་རི་བནྡྷེ། ཨ་ཐང་ཧ་བ་ར་བ་ན་ས་བ་བ་ཏི་བནྡྷེ། ན་བོ་ལ་ཁ་ཀ་པ་རུཾ་བནྡྷེ། ས་བ་སུ་མེ་ར་པར་བ་ཏ་བནྡྷེ། བནྡྷེ་བནྡྷེ་སཻ་ཨ་ད་བ་ར་ཧ་ན། དེ་ཁུ་མ་ཏེ་རེ་མནྟྲ་ཀི་སཀྟི། མི་རི་བཀྟི། ཕོ་རེ་མནྟྲ་ཨི་ཤཝ་རོ་བ་ཙ། ཞེས་པའི་སྔགས་འདི་དག་པར་བཟུང་ལ་ཐོག་མར་ཉི་ཟླ་གང་རུང་གཟའ་འཛིན་ཡོད་དུས་མདུན་དུ་མར་མེ། སྤོས། ལུག་ཤ་ཚོན་ཙམ། ཆང་ཕུད་བཅས་བཟའ་འཛིན་མ་གྲོལ་བར་བསྙེན་པ་གང་འགྲོ་སྐྱེལ། དེ་རྗེས་རང་བསྲུང་ན་གང་དགོས་དུས་བཟླས་ཤིང ། གཞན་འདྲེ་གདོན་གྱིས་གནོད་པ་ལ་ཕུས་བཏབ་པས། ཤ་ཟ། ཡི་དགས། འབྱུང་པོ་སོགས་འདྲེ་གདོན་གང་གིས་ཀྱང་གླགས་མི་རྙེད་དོ། །ཡང ། ཧ་ཐ་ཁ་པ་ར་ཀ་རེ་རེ་ཀ་པ་ལ། མ་རི་མས་ནི་ཧཱུཾ་ཕི་རུཾ་མོ་ཧི་ར་ཁོ་མ་ཧ་མ་གཱ་ལ། ཨ་དེ་བཻ་རོཾ་ཀི་རཀྵ། ཨུ་ཀ་དེ་བཻ་རོཾ་ཀི་རཀྵ། ཡཾ་ག་རཀྵེ་ཡཾ་གི་ཤཝ་ར། ཀ་ཡ་ཡ་རཀྵེ་བྷཻ་ས་སུ་ར། སི་སར་རཀྵེ་སཱི་ཏ་ལ། ཧི་ཡོ་རཀྵེ་བ་མ་ཧ་རེ། བ་མ་རཀྵེ་ཀྵེ་ཏྲ་བཱ་ལ། ན་པོ་ན་རི་བ་ཧ་ཏྲ་ཀོ་ཋ་ར་རཀྵེ་བ་ར་པི་ར། ཀཽ་ན་བི་ར་ཧ་ནུ་མིནྟ་བི་ར། རཱ་ཛ་བཱ་མ་ཙནྡྲ་ཀེ་པ་ཡི་ཀ། མ་ཧེ་ཤ་ཀུཥྛི་ཀི་བ་ཧ་ན། སཱི་ཏཱ་ཀེས་ཧོ་ད་ལ། མེ་རི་བྷཀྟེ། ཀུ་རུ་ཀི་སཀྟི། ཕོ་རི་མནྟ་ཨི་ཤཝ་རོ་བ་ཙ། ཞེས་པ་འདི་དག་བར་བཟུང་ལ་རབ་མིག་དམར་གཟར་ནམ་ཡོད་དང ། དེ་མིན་ཟླ་བ་རེར་མིག་དམར་ཡོད་པའི་ཚེས་གྲངས་གང་ཡིན་ལ་དོ་མ་སྟོངས་ཙམ་རེ་ཁྲུས་བྱས། མདུན་དུ་རས་གཙང་མའི་སྟེང་གྲོ་ཕྱེ། མར། དཀར་གསུམ་བསྲེས་པའི་རིལ་བུ་བྱ་སྒང་ཙམ་གཅིག་།དེའི་སྟེང་དུ་སིནྡྷཱུ་ར་ཚོན་ཙམ། མར་མེ། སྤོས་དང་བཅས་བཞག་ལ་ཚར་གསུམ་ལས་མི་ཉུང་བའི་གང་ལྕོགས་རེ་བཟླས་ན་གཞན་གྱིས་སྔགས། རྫས། ཏིང་ངེ་འཛིན་སོགས་
༄#། །ཀྱི་སྒོ་ནས་ངན་བྱེད་བྱས་པ་ཐམས་ཅད་བྱེད་པོ་ཁོ་རང་ལ་བཟློག་པ་ལས་གླགས་རྙེད་པར་མི་འགྱུར་ཞིང་འདྲེ་གདོན་གྱི་གནོད་པའང་མི་འབྱུང་ངོ ། ། དེ་ཤ་གུ་རུ་གུམ་པུརྦདེ་ཤ་ཡོ་གི་ནི། པ་ཚི་མ་ནིར་དྷ་ར། བནྡྷོ་ཁཾ་ཌ་པཱ་ཧཾ་པྲ་དྷ་ར། ཨི་ཤཝ་ར་མ་ཧཱ་དེ་བ། རུ་པེ་པཱ་ཨུ། སོ་དུ་རི་མེ་ཨཾ་ག་ཙ་དྷ་ལི། ཨ་ཏྲ་ཨ་ཏྲ་དྷ་ད་ཅུ་རི། རུ་ཨཾ་བཾ་ཧོ་ཡི། ཀམ་རུ་ཀ་མི་ཀྵ་ཀི་དོ་ཧ་ཡི། རུ་ཨཾ་བཾ་ག་དོ་ཧ་ཡི། གྲ་རུ་གོཪྵ་ན་ཐ་ཀི་ཧ་ཡི། མེ་རི་བྷ་ཀྟི་གུ་ར་ཀི་སཀྟི། ཕུ་ར་མནྟྲ་ནེ་ན་ཡོ་གི་ནཱི་བ་ཙ། ཞེས་པའི་སྔགས་འདི་ཡིད་ལ་བཟུང་ལ་གཡུལ་ལ་འཇུག་སྐབས་སོགས་ནམ་དགོས་དུས་ཚར་བདུན་པ་གྲངས་ནས་ལས་གང་རིགས་ལ་བཏབ་སྟེ་ལུས་ལ་བྱུགས་ན་གྲི་རིང་ཐུང་གང་གིས་ཀྱང་གླགས་མི་རྙེད་དོ། །ུ་ཧི་པ་ར་བ་ཏི་ལ་ཀ་རི་མངྒ་ཡི། སོ་ལ་ཀ་རི་ཧཾ་མ་ཙུ་ལེ་ལཻ། ཛ་ར་ཐཾ་བུ། ཛ་ར་ཀ་ཐཾ་བུ། ཐཾ་བུ་མ་གལ་ཏ་ཐི། ཛ་ར་ཏ་ཏེ་ལ་གི་།ཧུ་ཀ་ར་ཡི་ཐམ་བུ།་ཏྲཾ་བི་ས་ཀ་རམ་ས་ཁི། ཛོ་ཧཱུཾ་ཛེ་ཧཱུཾ་ཏེ་ལ་བ་རེ། སི་ར་ཨུ་བ་ར་གཽ་ར་ཀེ་སི་ར་ས་ཐི། ཕུ་རེ་མནྟྲ་ཨེ་ཤཝ་རོ་བ་ཙ། སྔགས་འདི་བློ་ལ་བཟུང་ལ། སྣུམ་དཀྲུག་སྐབས་ཚར་བདུན་བགྲངས་ལ་ལག་པར་ཕུས་བཏབ་ལ་དཀྲུགས་པས་ལག་པ་ཚིག་པར་མི་འགྱུར་རོ། །ཧ་ཐ་བ་ཁ་རུ། མུ་ཁ་དོ་ཨུཾ། ཌེ་ཌ་ཀེ་བ་དུཾ་བ་ར། རཱ་ཛ་ཧེ་ཀ་རུཾ་བར་ཛ། ན་ཧཱ་ཐ་ཧེ་རྒ་རུཾ་སེ་ཨ་ར་ས་དུ་བ་ཨེ་པཻ་ཐུཾ། སིངྒ་བྷ་ནི་ཀ་སུཾ། ཙིཏྟོ་ཀ་རེ་ཛི་བྱ་ཙ་རེ། བ་ཐུཾ་ན་ར་སིངྒ་ཛི་གི་མུ་ཧ་ནི། ར་ཧི་ཛ་ག་ཏྲ་མོ་ཚ་མི། དེ་ཥོ་ན་ར་སིངྒ་ཀི་སཀྟི། མེ་རེ་བྷཀྟི། ཕུ་རེ་མནྟྲ་ཨི་ཤཝ་རོ་བ་ཙ། ཞེས་བའི་སྔགས་འདི་ཚར་བདུན་བགྲངས་ལ་ཆུར་ཕུ་བཏབ་ནས་ཁ་ལག་གདོང་གསུམ་བཀྲུས་ཏེ་ཕྱིན་ན། རྒྱལ་པོ། རྒྱལ་ཕྲན། བློན་པོ། །གཙོ་བོ་སོགས་ཀྱིས་དགྱེས་པར་བརྩེགས་སྐྱོང་ཞིང་རང་གི་འདོད་པ་གང་ཡིན་སྟེར་ལ་གང་སྨྲས་བདེན་པར་འཛིན་ནོ། །ཀ་ཐུ་སུ་བ་རི་ཙུ་ཎ། ཨི་སྟི་ཀ་པེ་ཐ་དྷེ། པ་ཋོ་ཧོ་ཨི་ལ་གེ་མ་རེ་ཏོ་མ་ས་ན་ཧོ་ཨི། ཨི་ཛ་ག་ཛི་བཻ་ཧོ་ཏ་ར་བ་ཙ་ཏྲེ། མེ་རི་བྷ་ཀྟི་གུ་ར་ཀི་སཀྟི། ཕུ་རེ་མནྟྲ་ཨི་ཤཝ་རོ་བ་ཙ། ཞེས་པའི་སྔགས་འདི་གོ་ཡུ་ཕྱེ་མར་བཏབ་པ་ལ་ལན་སུམ་ཅུ་བགྲངས་ཏེ་བསྔགས་ལ་ཛ་ར་བཏབ་སྟེ་བུད་མེད་གང་འདོད་ལ་སྦྱིན་ན་རང་གི་
