༄།།རྒྱ་གར་སྐད་དུ།ཀྲྀཥྞ་ཡ་མཱ་རི་རཀྟ་ཡ་མཱ་རི་ཤྩ་པཱུ་ཛ་བི་དྷིཿ །བོད་སྐད་དུ།གཤིན་རྗེ་གཤེད་ནག་པོ་དང་དམར་པོ་མཆོད་པའི་ཆོ་ག།དཔལ་གཤིན་རྗེ་གཤེད་ལ་ཕྱག་འཚལ་ལོ།།དེ་ནས་བཟླས་པ་རྗེས་སུ་དྲན་པར་བྱ་སྟེ།ོཾ་ཧྲཱིཿ ཥྚྲྀཿ ཝི་ཀྲྀ་ཏཱ་ན་ན་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ།སྙིང་པོའི་སྔགས་སོ།།ོཾ་ཧྲཱིཿ ་ཀཱ་ལ་རཱུ་པ་ཧཱུཾ་ཁཾ།ཉེ་བའི་སྙིང་པོའོ།།ོཾ་ཧྲཱིཿ ་བྷྲཱུཾ་ཤྲཱུཾ་ཧཱུཾ་ཀཿ ་སྙིང་པོ་ཕྱི་མའོ།།ོཾ་ཧྲཱིཿ ་ཧཿ ་བྷོ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་གཙྪ་ཨཱ་གཙྪ།སྨཱཏ་པཱུ་ཛ་པྲ་ཏི་གྲྀ་ཧྞ་པྲ་སཱ་དྷཾ་མེ་དྷི་མན་ཀུ་རུ་སཝཱ་ཧཱ།སྤྱན་དྲང་བའོ།།ོཾ་ཧྲཱིཿ ་ཥྚྲྀཿ ་ཨཱ་མཆོད་ཡོན་ནོ།།ོཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ།ཞབས་བསིལ་བའོ།།ོཾ་ཝི་ཀྲྀ་ཏཱ་ན་ན་དུཥྚ་སཏྭ་ད་མ་ཀ་གཿ ་གཿ ་དྲིའི་སྔགས་སོ།།ོཾ་ཀུ་མཱ་ར་རཱུ་པི་ཎི་ཛཿ ་ཛཿ ་ཧཱུཾ་ཕཊ།མེ་ཏོག་དབུལ་བའི་སྔགས་སོ།།ོཾ་ཧྲཱིཿ ཧཿ ཧེ་ཕཊ།བདུག་སྤོས་སོ།།ོཾ་ཀཱ་ལ་རཱུ་པ་སརྦ་དུཥྚཱཾ་སཏྭ་ད་མ་ཀ་མ་ཧཱ་བྷཱུ་ཏ་པ་ཏི།།ི་མཾ་བ་ལིཾ་གྲྀཧྞཱ་པ་ཡ་སཝཱ་ཧཱ།གཏོར་མའི་སྔགས་སོ།།ོཾ་དཱིཔྟ་ལོ་ཙ་ནི་ཝི་ཀྲྀ་ཏཱ་ན་ན་མ་ཧཱ་ཊྚཱ་ཊྚ། ཧཱ་ས་ནཱ་དི་ནི་དཱིཔྟ་དཱིཔྟཱ་ཡ་སཝཱ་ཧཱ།མར་མེའི་སྔགས་སོ།།ོཾ་ཛ་ཡ་ཛ་ཡ།སུ་ཛ་ཡ།བི་ཤཝ་རཱུ་པ་དྷ་རི་གཙྪ་གཙྪ།སཝ་བྷཱ་བ་ནཱཾ་ཀྲྀ་ཏེ་ན།ཏ་ཐཱ་ག་ཏཾ་ཤྩ་སརྦ་བུདྡྷཾ་ཤྩ་བི་ས་ར་ཛ་ཡ་མ་མ་ཀཱརྱཾ་ཀུ་རུ་སཝཱ་ཧཱ།གཤེགས་སུ་གསོལ་བའི་སྔགས་སོ།།གཤིན་རྗེ་གཤེད་ནག་པོ་དང་གཤིན་རྗེ་གཤེད་དམར་པོའི་མཆོད་པའི་ཆོ་ག།སློབ་དཔོན་ཆེན་པོ་པཎྜི་ཏ།དཔལ་འཛིན་གྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།།།།པཎྜི་ཏ་ཉི་མའི་དབང་པོའི་འོད་ཟེར་དང།ཆག་ཆོས་རྗེ་དཔལ་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།།
