༄།།རྒྱ་གར་སྐད་དུ།ཱརྱ་ར་པ་ཙ་ན་པཱུ་ཛ་བི་དྷི་ནཱ་མ།བོད་སྐད་དུ།འཕགས་པ་ཨ་ར་པ་ཙ་ནའི་མཆོག་པའི་ཆོ་ག་ཞེས་བྱ་བ།འཕགས་པ་ཨ་ར་པ་ཙ་ན་ལ་ཕྱག་འཚལ་ལོ།།སྒྲུབ་ཐམས་སུ་བཤད་པའི་རིམ་པས་དེ་ལྟར་བླ་མའི་མཎྜལ་སྔོན་དུའོ།།ཤིན་ཏུ་མཉམ་པར་
གཞག་ལ་འགྲོ་ལ་ཕན་པའི་སེམས་ཀྱིས་མཆོད་པར་བྱའོ།།ོཾ་ཨཱཿ ཧཱུཾ་གནས་དང་བདག་དང་རྣལ་འབྱོར་བསྲུང་བའོ།།དེ་ནས་ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏཱ་སུ་ལ་ལི་ཏ།ལ་ལི་ཏཻ་རྣ་ནཱ་མཱ་མི་བྷ་ག་བནྟཾ་ཨ་ར་པ་ཙ་ན་ནཱ་ཐཱ།ཇཿ ་ཧཱུཾ་བཾ་ཧོཿ ་པྲ་ཏཱི་ཙྪ་མཾ་ཀུ་སུ་མཱ་ཉྫ་ལི་ནཱ་ཐ་ཧོཿ །ོཾ་ཨཱཿ ཧྲཱིཿ ་པྲ་བ་ར་སཏྐཱ་ར་ཡ་ཨ་ར་པ་ཙ་ནཱ་ཡ་པཱ་དྱཾ་པྲ་ཏཱི་ཙྪ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ཨཱཿ ཧྲཱིཿ པྲ་བ་ར་སཏྐཱ་ར་ཡ་ཨ་ར་པ་ཙ་ནཱ་ཡ་པྲོ་ཀྵ་ཎ་པྲ་ཏཱིཙྪ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ཨཱཿ ་ཧྲཱིཿ པྲ་བ་ར་སཏྐཱ་ར་ཡ་ཨ་ར་པ་ཙ་ནཱ་ཡ་ཨརྩྪ་མ་ནཾ་པྲ་ཏཱིཙྪ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ཨཱཿ ་ཧྲཱིཿ པྲ་བ་ར་སཏྐཱ་ར་ཡ་ཨ་ར་བ་ཙ་ནཱ་ཡ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་མཉྫུ་གྷོ་ཤ་ཡ་དྷཱིཿ ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་ཛཱ་ལི་ནཱི་པྲ་བྷཱ་ཡ་རཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མ་ཤྩནྡྲ་པྲ་བྷཱ་ཡ་པཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱུཾ།ོཾ་ན་མཿ ཀེ་ཤི་ནྱེ་ཙཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མཿ ་ཨུ་པ་ཀེ་ཤི་ནྱེ་ནཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མཿ ་ཨ་ཀྵོ་བྷྱཱ་ཡ་ཧཱུཾ།ོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་བེ་རོ་ཙ་ནཱ་ཡ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་རཏྣ་སཾ་བྷ་བཱ་ཡ་ཏྲཱཾ།ོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་ཨ་མི་ཏཱ་བྷཱ་ཡ་ཧྲཱིཿ ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མ་ལོ་ཙ་ནཱ་ཡེ་ལཱཾ་ཨོཾ་ཨོ།བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མོ་ལཱ་མེ་ཀྱེ་མཱཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ན་མཿ ་པཎྜ་རཱ་ཡཻ་པཱཾ་ཨོཾ་བཛྲ་པུཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།། ཨོཾ་ན་མ་སྟཱ་རཱ་ཡཻ་ཏཱཾ་ཨོཾ་བཛྲ་པེཥྤེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་ཨཱཿ ཧཱུཾ།འད་ཡི་རིམ་པས།ོཾ་བཛྲ་དྷཱུ་པེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་བཛྲ་དཱི་པེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་བཛྲ་གནྡྷེ་ཧཱུཾ་སཝཱ་ཧཱ།ོཾ་བཛྲ་ནཻ་བི་དྱེ་ཧཱུཾ་སཝཱ་ཧཱ།དེ་རྣམས་ཀྱི་མདུན་དུ་འདོད་པའི་ཡོན་ཏན་ལྔ་པོ་རྣམས་ཉེ་པར་ཡང་དག་པར་དབུལ་བར་བྱའོ།།ོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པཱུ་ཛ་མ་གྷ་པྲ་ས་ར་ས་མུ་དྲ་སྥར་ཎ་ཧི་མཾ་ག་གན་ཁཾ་ཧཱུཾ་ཕཊ།མཆོད་པ་བྱིན་གྱིས་བརླབས་པར་བྱའོ།།དངོས་པོ་མེད་ཀྱང་ཡིད་ལས་བྱུང་བའི་མཆོད་པ་དབུལ་ལོ།།ན་མཿ ་སརྦ་བུདྡྷ་བོ་དྷི་སཏྭཱ་ནཱཾ་སརྦ་ཐཱ་ཁཾ་ཨུངྒ་ཏེ་སྥར་ཎ་ཧི་མཾ་ག་ག་ན་ཁཾ་ས་མནྟ་སཝཱ་ཧཱ།ཡིད་ཀྱིས་མཆོད་པ་བྱིན་གྱིས་བརླབ་པའོ།།རི་བོང་འཛིན་པ་ལྟ་བུ་ཡི།།ཅེས་པ་ལ་སོགས་པ་དང།རིན་ཆེན་གསུམ་རྣམས་ཕྱག་འཚལ་ལོ།།མཆོད་པའང་རིན་ཆེན་གསུམ་པོ་ལ།།སྡིག་པ་གང་རྣམས་བརྗོད་པར་བརྗོད་པར་བགྱི།བསོད་ནམས་ཀུན་ལ་རྗེས་སུ་དགའ།།ཆོས་ཀྱི་འཁོར་ལོ་རབ་བསྐོར་ཞིང།།མྱང་ངན་འདའ་རྟག་ཏུ་བཞུགས།།བདག་གིས་སྦྱིན་སོགས་བསོད་
༄།།ནམས་ཀྱིས།།བདག་ལ་ཀུན་རྟོག་ཡོངས་སུ་བསྒྱུར།།ཞེས་པ་རྣམ་པ་བདུན་བླ་ན་མེད་པའི་མཆོད་པ་བྱས་ནས་དབང་བསྐུར་བའི་ཕྱིར།སྤྱན་མ་ལ་སོགས་པའི་ལྷ་མོས་མངོན་པར་དགའ་བས་དབང་བསྐུར་བའོ།།རང་གི་རིགས་ཀྱི་ངོ་བོར་བྱས་ནས་འདོད་པའི་ཕྱིར་བདག་ལ་དབང་བསྐུར་རོ།།གཙང་མའི་གནས་ཀྱི་ལྷ་རྣམས་ཀྱིས།།ཇི་ལྟར་དེ་བཞིན་གཤེགས་འཁྲུངས་ཚེ།།ཀུན་བཀྲུས་དེ་བཞིན་བདག་དབང་བསྐུར།།ོཾ་ཨཱཿ ་སརྦ་ཏ་ཐཱ་ག་རཱ་བྷི་ཥེ་ཀ་སམ་ཡ་ཤྲི་ཡེ་ཧཱུཾ།ོཾ་སརྦ་ཏ་ཐཱ་ག་ཏཱ་བྷི་ཥེ་བཛྲ་སཝ་བྷཱ་བ་ཨཱ་ཏྨ་ཀོཉ?ཧཾ།ཡི་གེའི་རང་བཞིན་འབྱུང་གནས་དང།།ལྷན་ཅིག་ཀུན་དུ་རྣམ་པར་རྟོག།།གྲངས་མེད་ཨཱ་ལི་ཀཱ་ལིའི་ཚོགས།།དེ་ཉིད་རྟོགས་པའི་རྣམ་པར་འགྲོ་།ོཾ་པ་དེ་པ་དེ་མ་ཧཱ་ཛྙཱ་ན་སརྦ་བུདྡྷ་མ་ཡཾ་བྷ་མེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧོཿ ་ཧོཿ ་ཧོཿ ་ཨ་ཁཾ་སཝཱ་ཧཱ།འདི་ཡིས་གསང་སྔགས་ཀྱི་ཕྲེང་བ་རྣམས་བྱིན་གྱིས་བརླབས་ལ་གསང་སྔགས་བཟླས་སོ།། གསང་སྔགས་བླ་མའི་ཉེ་བར་བཤད་པ་ལས་རྟོགས་པར་བྱའོ།།བཟླས་པ་བྱས་ནས་གཏོར་མ་བྱིན་གྱིས་བརླབ་པར་བྱ་སྟེ།ཡཾ་ཡིག་གིས་རླུང་གི་དཀྱིལ་འཁོར་རོ།།རྀ་ཡིག་གིས་མེའི་དཀྱིལ་འཁོར་རོ།།དེའི་སྟེང་དུ་ཀཾ་ཡིག་གིས་སྒྱེད་པུ་གསུམ་གྱི་སྟེང་དུ། ཡིག་ལས་སྐྱེས་པའི་སྣོད་དུ་འཆི་མེད་ལྔ་དང་རབ་ཏུ་གསལ་བ་ལྔ་པོ་ལྷན་ཅིག་བཞུ་ནས།འཆི་མེད་ཡེ་ཤེས་བཅུ་གནས་དབུལ་བར་བྱ་སྟེ།ོཾ་ཨ་ར་པ་ཙ་ནཱ་ཡ་ཨདྒ་པུ་སྟཱ་ཀ་ཧ་སྟཱ་ཡ་ཨི་མཾ་བ་ལིཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི།ཧ་ན་ཧ་ན།ད་ཧ་ད་ཧ།པ་ཙ་བ་ཙ།མ་མ་སརྦ་སཏྭཱ་ནཱ་ཉྩ་པྲ་ཏྱའ་རྠ་ཀ་ན་བནྡྷ་བནྡྷ་ནཱི་དྷཱིཿ ་ཤྲི་ཁམ། ར་བ་ཙ་ནཱ་ཡ་སཝཱ་ཧཱ།གཏོར་མའི་སྔགས་སོ།།ཆད་པ་དང་ལྷག་པ་དགང་བའི་ཕྱིར།ཡི་གེ་བརྒྱ་བ་བརྗོད་པར་བྱ་སྟེ།ོཾ་བཛྲ་སཏྭ་ས་མ་ཡ།མ་ནུ་པཱ་ལ་ཡ།བཛྲ་སཏྭ་ཏྭེ་ནོ་པ།ཏིཥྛ་དྲྀ་ཌྷོ་མེ་བྷ་བ།སུ་ཏོ་ཥྱོ་མེ་བྷ་བ།སུ་པོ་ཥྱོ་མེ་བྷ་བ། ནུ་རཀྟོམེ་བྷ་བ། སརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪ།སརྦ་ཀརྨྨ་སུ་ཙ་མེ་ཙིཏྟཾ་ཤྲེ་ཡཿ ་ཀུ་རུ་ཧཱུཾ།ཧ་ཧ་ཧ་ཧ་ཧོཿ ་བྷག་བཱན་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་མཱ་མེ་མུཉྩ་བཛྲཱི་བྷ་བ་མ་ཧཱ་སམ་ཡ་སྲྭ་ཨཱཿ ་བདག་ཅག་སེམས་ཅད་དོན་ཀུན་མཛད།།ཇི་ལྟར་རྗེས་རྟོགས་དངོས་གྲུབ་སྩོལ།།སངས་རྒྱས་ཡུལ་དུ་གཤེགས་པ་དང།།
