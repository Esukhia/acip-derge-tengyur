༄##། །རྒྱ་གར་སྐད་དུ། བ་ཤྱ་དྷི་ཀཱ་ར་མཉྫུ་ཤྲཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ལྷག་པར་དབང་དུ་བྱེད་པའི་འཇམ་དཔལ་་གྱི་སྒྲུབ་ཐབས། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་འཇམ་དཔལ་རྡོ་རྗེའི་སྐུར་བསྒྲུབས་ཏེ། །སྐུ་མདོག་དམར་པོ་འོད་ཟེར་འབར་བའི་ཚོགས། །དབྱངས་ཡིག་བཅུ་དྲུག་ཡང་དག་འཁོར་བ་ཡི། །སྙིང་གའི་པད་དབུས་རྡོ་རྗེ་བསམ་པར་བྱ། །བནྡྷུའི་མེ་ཏོག་རྣམ་པ་དང་མཚུངས་་སྐུར། །དམིགས་ནས་ངེས་པར་རྐང་དྲུག་བཤད་པ་ཡིས། །དབུགས་ནི་རླུང་གིས་མ་ལུས་དལ་བུ་ཡིས། །ངེས་པར་དྲངས་ནས་སྐྱེ་བའི་བསམ་པ་ཡང ། །བསྒྲུབ་བྱ་མདུན་དུ་གནས་ཤིང་ཐལ་མོ་སྦྱོར། །མི་གཟུང་གཉིད་ནི་ཡུར་བར་ཡང་དག་བསམ། །སྣ་བུག་ནས་ནི་དེ་བཞིན་བུང་བ་ཡིས། ། དེ་རྣམས་བསྒྲུབ་བྱ་ཉིད་ཀྱི་སྙིང་གར་ཞུགས། །སྙིང་གའི་པདྨ་ཀུན་ནས་བལྡག་ཅིང་རྙིངས། །དེ་ཡི་ཡིད་ཀྱི་སྦྲང་རྩི་འཐུངས་པ་ཡིས། །ངེས་ཕྱུང་དེ་ལས་སླར་ཡང་ཞུགས་གྱུར་ནས། །དེ་ལ་དེ་བཞིན་ངེས་པར་ཡིད་ཐིམ་བསམ། །རྣམ་པ་དེ་བཞིན་བརྟན་པ་དང་ལྡན་སེམས། །ཐུན་མཚམས་བཞི་་
