༄##་།།རྒྱ་གར་སྐད་དུ། ཨཱརྱ་མཉྫུ་ཤྲཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། འཕགས་པ་འཇམ་དཔལ་གྱི་སྒྲུབ་ཐབས། འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །རྣམ་དག་་འཇམ་དཔལ་སྒྲུབ་པའི་ཐབས། །ཤིན་ཏུ་གསལ་ཞིང་མདོར་བསྡུས་པ། །མི་ཤེས་རབ་རིབ་འཇོམས་པར་བྱེད། །དམ་པ་རྣམས་ཕྱིར་ལེགས་བརྗོད་བྱ། །སྟོང་པ་ཉིད་ཀྱི་རྗེས་སུ་དེ་བཞིན་དུ་སྙིང་གར་ཡི་གེ་མུཿ སེར་པོ་ཡོངས་སུ་གྱུར་པ་ལས། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ཞལ་གསུམ་ཕྱག་དྲུག་་
