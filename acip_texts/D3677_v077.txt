༄##། །རྒྱ་གར་སྐད་དུ། ཏཱ་རཱ་ཊོ་བ་བི་དྷིཿ ། བོད་སྐད་དུ། སྒྲོལ་མའི་ཁྲུས་ཀྱི་ཆོ་ག་།རྗེ་བཙུན་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །ཁྲུས་ཀྱི་ཆོ་ག་ལ་གསུམ་སྟེ། སྦྱོར་བ་དང ། དངོས་གཞི་དང ། རྗེས་སོ། །དང་པོ་ནི་གནས་ཡིད་དང་་མཐུན་པར་ཕྱག་དར་ལ་སོགས་པ་བྱས་ཏེ། བུམ་པ་མཚན་ཉིད་དང་ལྡན་པར་རྫས་གཞན་དང་མཐུན་པར་བསགས་ཏེ། ཤེལ་མེད་དུ་མི་རུང་མཆོད་པ་རྣམས་སྟ་གོན་བྱའོ། །དངོས་གཞི་ནི་སྔོན་དུ་གསུངས་པའི་རིམ་པ་སྟོང་པའི་ངང་ལས་སེམས་ཉིད་ཏཱཾ་དཀར་པོར་བལྟའོ། །དེ་ལས་འོད་་འཕྲོས་བསྡུས་ནས་བདག་ཉིད་སྒྲོལ་མ་མདོག་དཀར་མོ་པདྨ་དང་ཟླ་བ་ལ་རྡོ་རྗེ་སྐྱིལ་ཀྲུང་གིས་བཞུགས་པ། ཞལ་གཅིག་ཕྱག་གཉིས་མཉམ་གཞག་གི་སྟེང་ན་བདུད་རྩིའི་བུམ་པ་བསྣམས་པ། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་རྒྱན་དང་ལྡན་པ་འོད་འཕྲོ་བ་དང་བཅས་པའི་སངས་རྒྱས་ལྔའི་དབུ་རྒྱན་ཅན་དུ་་བསྐྱེད་པར་བྱའོ། །བུམ་པ་ཡང་མི་དམིགས་པ་ལས་ཕྱི་བུམ་པ་ལ་ནང་རིན་པོ་ཆེའི་གཞལ་ཡས་ཁང་བཅུད་བདུད་རྩིའི་རྒྱ་མཚོའི་དབུས་སུ་པདྨའི་པ་དྲའི་སྡོང་པོ་ལ་རང་འདྲ་བའི་དམ་ཚིག་སེམས་དཔའ་བསྐྱེད། དེའི་མདུན་ན་ལས་ཀྱི་ཁྲོ་བོ་བདུད་རྩི་འཁྱིལ་པ་དམར་སེར་བཞེངས་པ།
