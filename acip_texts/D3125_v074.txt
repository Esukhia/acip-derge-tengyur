༄##། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་མ་ཧཱ་པྲ་ཏི་སཱ་རཱ་ཡཻ་བིདྱཱ་བི་དྷི་ནཱ་མ། བོད་སྐད་དུ། འཕགས་མ་སོ་སོར་འབྲང་མ་ཆེན་མོའི་རིག་པའི་ཆོ་ག་ཞེས་བྱ་བ། འཕགས་མ་སོ་སོར་འབྲང་མ་ཆེན་མོ་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་ཐམས་ཅད་མངོན་དགའ་བའི། །ཕྱག་འཚལ་་སོ་སོར་འབྲང་མ་བཤད། །སེམས་ཅན་ཐམས་ཅད་རྣམ་འབྱུང་བའི། །དེ་ཡི་འཁོར་ལོ་བདག་གིས་བྲི། །གཙང་མ་མཉམ་པར་གཞག་ཅིང་གཏོར་མ་ལ་སོགས་བྱིན་ལ། བསྐུམ་ཁྲུ་གང་གི་ཚད་ཀྱི་གྲོ་གའམ་རས་ལ་སོགས་པའི་གཞི་ལ། དང་པོ་རེ་ཞིག་སོར་བཞིའི་ཚད་ཀྱི་རིམ་པ་་གཅིག་བྲིའོ། །དེའི་དབུས་སུ་ཟླ་ཚེས་ཐིག་ལེས་བརྒྱན་པའི་ཧཱུཾ་བརྒྱད་ཀྱིས་བསྐོར་བའི་པྲཾ་ཡིག་གོ། །དེའི་ནང་དུ་སྔགས། ཨོཾ་མ་ཎི་དྷ་རི་བཛྲི་ནི་མ་ཧཱ་པྲ་ཏི་སཱ་རེ་དྷ་རེ་ཀསྱཾ་རཀྵཾ་ཀུ་རུ་ཤཱནྟི་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། དེའི་ཕྱིར་སོར་ཕྱེད་ཀྱི་རིམ་པ་གཉིས་ལ། དེ་ལ་དང་པོ་རིམ་པ་གཅིག་གི་་སྔགས་ནི། ཨོཾ་མ་ཎི་ཞེས་པ་སྔར་གྱི་གཅིག་གོ། །གཉིས་བ་ནི་དེ་ལྟར་རིམ་པར་བྱའོ། །དེའི་ཕྱི་རོལ་དུ་སོར་གསུམ་གྱི་རིམ་པ་གཅིག་གོ། །དེ་ལ་ཕྱོགས་བཞིར་རི་མོ་གཉིས་པ་པདྨ་འདབ་མ་བཞི་པའི་མདུན་ལ་སོགས་པའི་ཕྱོགས་ཀྱི་རིམ་པར་རི་མོ་གཉིས་ཀྱི་མཚམས་སུ་སྔགས་ནི། 
༄#། །ོཾ་བི་མ་ལེ་བི་པུ་ལེ་ཛ་ཡ་བ་རེ་ཨ་མྲྀ་ཏེ་དྷ་ར་རཀྵ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། གཡས་སུ་ཨོཾ་ཨ་མྲྀ་ཏེ་བི་ལོ་ཀི་ནི་གརྦྷ་སཾ་རཀྵ་ཎི་ཨཱ་ཀཪྵ་ཎི་དྷ་ར་ར་ཀ་སྱ་རཀྵ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། རྒྱབ་ཏུ། ཨོཾ་བྷཱ་ར་བྷཱ་ར། སཾ་བྷཱ་ར་སཾ་བྷཱ་ར། ཨིནྡྲཱི་ཡ་བ་ལ་བི་ཤོ་དྷ་ནི་དྷ་་ར་ཀ་སྱ་རཀྵཾ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། གཡོན་དུ་ཨོཾ་ཨ་མྲྀ་ཏ་བ་རེ་བ་ར་བ་ར་པྲ་བ་ར་བི་ཤུདྡྷེ་ར་ཀ་སྱ་རཀྵ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། འདབ་མ་རྣམས་ཀྱི་དབུས་སུ་ཧཱུཾ་ཡིག་བརྒྱད་ཀྱིས་བསྐོར་བའི་པྲཾ་ཡིག་གི་ནང་དུ། ཨོཾ་མ་ཎི་དྷུ་རི་ཞེས་པ་ལ་སོགས་པའི་སྔགས་བྲི་བར་་བྱའོ། །ཕྱོགས་བྲལ་བཞིར་རྩེ་གསུམ་པའི་རྡོ་རྗེ་ཕྱེད་པའི་ནང་གི་མེ་ལ་སོགས་པའི་མཚམས་ཀྱི་རིམ་པར་སྔགས་ནི། ཨོཾ་བི་སྥོ་ཊ་བཛྲི་ནི་དྷ་ར་ཀ་སྱ་རཀྵ་ཤཱིནྟཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། བདེན་བྲལ་དུ། ཨོཾ་ཨ་མྲྀ་ཏ་བཱ་རེ་དྷ་ར་ཀ་སྱ་རཀྵཾ་ཤཱནྟིཾ་པུཥྚི་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། རླུང་དུ། ཨོཾ་བི་མ་་ལེ་དྷ་ར་ཀ་སྱ་རཀྵཾ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། དབང་ལྡན་དུ། ཨོཾ་ཛ་ཡ་བཱ་རེ་དྷ་ར་ཀ་སྱ་རཀྵཾ་ཤཱིནྟཾ་པུཥྚིཾ་ཀུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། པདྨའི་འདབ་མ་དང་རྡོ་རྗེ་ཕྱེད་མ་རྣམས་ཀྱི་མཚམས་བརྒྱད་རྣམས་སུ། མེའི་ཕྱོགས་ཀྱི་རིམ་པར་སྔགས་ནི། ཨོཾ་སུཾ་བྷ་བཛྲི་ཎི་དྷ་ར་ཀ་སྱ་རཀྵཾ་ཤཱནྟིཾ་པུཥྚིཾ་ཀུ་་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཝཱ་ཧཱ། ཞེས་པའི་སྔགས་གཉིས་སམ་གཅིག་ནང་དུ་བྲི་བར་བྱའོ། །དེའི་ཕྱི་རུ་སོར་ཕྱེད་དང་གསུམ་གྱི་ཚད་ལ་རིམ་པ་བཞི་བྱའོ། །དེ་ལ་རིམ་པ་དང་པོ་ལ་ཧྲཱིཿ ཡི་གེ་ཕྲེང་གཅིག་བྲིའོ། །དེའི་ཕྱིའི་རིམ་པ་ལ། ཨོཾ་ཨ་པ་རཱ་ཛི་ཏེ་མ་ཧཱ་དེ་བི་ཏྲཊ། ཨོཾ་ཀཱ་ལ་ཀཎྜ་ཀྲི་ཏ། པདྨ་་ཀུཎྜ་ལི་པཾ། ཨོཾ་པུཥྤ་དནྟི་ནི་ཧཱུཾ། ཨོཾ་མ་ལྱཾ་ཀུ་ཤི་ཧཱུཾ། ཨོཾ་བཛྲ་སྲིཾ་ཁ་ལེ་ཧཱུཾ། ཨོཾ་ཤཝེ་ཏེ་བཾ། ཨོཾ་མ་ཧཱ་ཀཱ་ལི་ཧྲཱིཿ ། ཨོཾ་བཛྲ་པཱ་ཎི་ཧེ། ཨོཾ་བཛྲ་པཱ་ཎི་བྲྀ་ཏ། ཨོཾ་བཛྲ་པཱ་ཎི་མ་ཧཱ་བ་ལེ་དྷིཿ ། ཨོཾ་བཛྲ་མཱ་ལེ་མཾ། ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལ་སཝཱ་ཧཱ། ཨོཾ་ཛྭ་ལ་ཛྭ་ལ་ཧཱུཾ་ཕཊ་སཝཱ་ཧཱ། ཨོཾ་ཊཀྐི་ཧཱུཾ་ཛཿ སཝཱ་ཧཱ། དེའི་ཕྱིའི་རིམ་པ་ལ་ཨོཾ་་ཙུནྡི་ཧྲཱིཿ ། ཨོཾ་སུརྞྞ་ཀེ་ཤི་ཀྲྀ་ཏ། ཨོཾ་པིཾ་ག་ལེ་ལཾ། ཨོཾ་བི་དྱུནྨཱ་ལི་ནི་ཛཿ ཧཱུཾ། ཨོཾ་ཤཱནྟིཾ་ཀ་རི་ཧཱུཾ། ཨོཾ་པུཥྚིཾ་བ་ལ་བརྟ་ནི་ཧཿ ། ཨོཾ་སུམྦྷ་ཀ་རི་ཧཱུཾ། ཨོཾ་མ་ཎི་དྷ་རི་ཧཱུཾ། ཨོཾ་རཏྣ་མུ་ཀུ་ཊ། མཱ་ལ་དྷ་རི་ཧྲཱིཿ ། ཨོཾ་མ་ཧཱ་པྲ་ཏི་ས་རི་པྲཾ། ཨོཾ་དྷུ་རུ་དྷུ་རུ། མུ་རུ་མུ་རུ། ཧུ་རུ་ཧུ་རུ། མེ་རུ་མེ་རུ། ཧཱུཾ་ཧཱུཾ། ཨོཾ་བ་ལ་བརྡྷ་ནི་་
