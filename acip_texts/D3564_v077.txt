༄##། །རྒྱ་གར་སྐད་དུ། ཀུ་རུ་ཀུལླེ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། རིགས་བྱེད་མའི་སྒྲུབ་ཐབས། རིགས་བྱེད་མ་ལ་ཕྱག་འཚལ་ལོ། །དང་པོར་སྟན་་བོད་བ་ལ་ཉེ་བར་འདུག་སྟེ། རིགས་བྱེད་མའི་སྐུར་བསྒོམས་ནས་སེམས་ཅན་ཐམས་ཅད་དབང་དུ་བྱ་བ་བརྩམ་པར་བྱའོ། །རང་གི་སྙིང་གར་ཉི་མའི་དཀྱིལ་འཁོར། དེའི་སྟེང་དུ་ཏཱཾ་ཡིག་དམར་པོ། དེ་ལས་འོད་ཟེར་ལྔ་བྱུང་སྟེ་དེ་བཞིན་གཤེགས་པ་རྣམས་ཡང་དག་པར་མཆོད་ནས། དཀོན་་
